    function formatPrice(price) {

    function applyFilters() {
      let filtered = bikes.slice();

      const typeValue = filterType.value;
      const useValue = filterUse.value;
      const electricOnly = filterElectricOnly.checked;
      const maxPrice = filterMaxPrice.value ? parseInt(filterMaxPrice.value, 10) : null;

      if (typeValue) {
        filtered = filtered.filter(b => b.type === typeValue);
      }

      if (useValue) {
        filtered = filtered.filter(b => b.use === useValue);
      }

      if (electricOnly) {
        filtered = filtered.filter(b => b.electric);
      }

      if (!isNaN(maxPrice) && maxPrice !== null) {
        filtered = filtered.filter(b => b.price <= maxPrice);
      }

      const sortValue = sortBySelect.value;
      if (sortValue === "price-asc") {
        filtered.sort((a, b) => a.price - b.price);
      } else if (sortValue === "price-desc") {
        filtered.sort((a, b) => b.price - a.price);
      } else if (sortValue === "brand") {
        filtered.sort((a, b) => (a.brand + a.name).localeCompare(b.brand + b.name, "fr"));
      }

      renderBikes(filtered);
    }

    function renderBikes(list) {
      bikesGrid.innerHTML = "";

      if (list.length === 0) {
        resultsEmpty.style.display = "block";

      resultsEmpty.style.display = "none";
      resultsCount.textContent =

        image.className = "bike-card__image";
        image.textContent = bike.imageText;

        const body = document.createElement("div");
        body.className = "bike-card__body";

        const name = document.createElement("div");
        name.className = "bike-card__name";
        name.textContent = bike.name;

        const brand = document.createElement("div");
        brand.className = "bike-card__brand";
        brand.textContent = bike.brand;

        const meta = document.createElement("div");
        meta.className = "bike-card__meta";
        const typeLabels = {
          route: "Route",
          gravel: "Gravel",
          vtt: "VTT",
          urbain: "Urbain / trekking"
        };
        const useLabels = {
          endurance: "Endurance / longue distance",
          montagne: "Montagne / cols",
          trail: "Trail / all-mountain",
          urbain: "Urbain / déplacements",
          voyage: "Voyage / bikepacking"
        };
        const electricLabel = bike.electric ? "Électrique" : "Musculaire";
        meta.textContent =
          typeLabels[bike.type] + " · " +
          useLabels[bike.use] + " · " +
          electricLabel;

        const price = document.createElement("div");
        price.className = "bike-card__price";
        price.textContent = "dès " + formatPrice(bike.price);

        const tagsContainer = document.createElement("div");
        tagsContainer.className = "bike-card__tags";

        if (bike.electric) {
          const tagElectric = document.createElement("span");
          tagElectric.className = "tag tag--green";
          tagElectric.textContent = "Électrique";
          tagsContainer.appendChild(tagElectric);
        }

        if (Array.isArray(bike.tags)) {
          bike.tags.forEach(t => {
            const tag = document.createElement("span");
            tag.className = "tag";
            tag.textContent = t;
            tagsContainer.appendChild(tag);
          });
        }

        const cta = document.createElement("div");
        cta.className = "bike-card__cta";
        cta.textContent = "Détails & dispo en magasin";

        body.appendChild(name);
        body.appendChild(brand);

        body.appendChild(price);
